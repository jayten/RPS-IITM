  \textbf{$c_p$} \> heat capacity at constant pressure \\
  \textbf{$d_w$} \> distance from nearest wall surface \\
  \textbf{$Pr$} \> Prandtl number, taken as constant 0.72 \\
  \textbf{$Pr_t$} \> turblent Prandtl number taken as constant 0.9 \\
  \textbf{$k$}  \> turbulent kinetic energy \\
  \textbf{$H$} \> enthalpy \\
  \textbf{$R_k$} \> Cell-based residual for transport equation of k; normalized with $\text{a}_{ref}^4/\nu_{ref}$ \\
  \textbf{$R_\omega$} \> Cell-based residual for transport equation of $\omega$; normalized with $\text{a}_{ref}^4/\nu_{ref}^2$ \\
  \textbf{$||R||_k, ||R||_\omega$} \> L2-norm of non-dimensional residual of all cell for $k \text{and } \omega$ eq. respectively \\
  \textbf{$S$}   \> Absolute value of strain rate, $(2S_{ij}S_{ij})^{1/2}$ \\
  \textbf{$S_{ij}$}   \> strain rate tensor, $0.5(\partial u_i/\partial x_i + \partial u_j/\partial x_i)$ \\
  \textbf{$T$} \> local temperature \\
  \textbf{$U$} \> local velocity magnitude \\
  \textbf{$u,v,w$} \> component of local velocity in the x, y and z direction \\
  \textbf{$X,Y,Z$} \> cartesian coordiantes \\
  \textbf{$\mu$}   \> molecular viscosity \\
  \textbf{$\mu_t$}   \> eddy viscosity \\
  \textbf{$\rho$}   \> density \\
  \textbf{$\gamma$}   \> intermittency \\
  \textbf{$\Omega$}   \> Absolute value of vorticity, $(2\Omega_{ij}\Omega_{ij})^{1/2}$ \\
  \textbf{$\Omega_{ij}$}   \> vorticity tensor, $0.5(\partial u_i/\partial x_i - \partial u_j/\partial x_i)$ \\
  \textbf{$\omega$}  \> specific turbulence dissipation rate \\
